rarity(2)

title("Mercenaries")
types(loot, summon)
cost(1)
$Range=2
$cardnumber=0
trigger(onPlay,do(setVar(self,cardnumber,(count(Hand,loot)))))
#trigger(onPlay,enemyDo(say(Player,$cardnumber)))
trigger(onPlay, if(($cardnumber > 0),(do(summon("Shellsworn",(any),$Range)))))
trigger(onPlay,foreach(Loopcard,(select(Hand, (hastype(self, loot)),"-",all)),(do (purge($Loopcard )))))
trigger(onPlay, do(addStatus(expire,$cardnumber, (do(selectTiles((all,(any),"name:Shellsworn"),$Range,Player))))));
trigger(onPlay, do(addVar(self,removecount,$cardnumber)));
removetrigger(endofturn, true, 5);
rarity(3)
text("Purge all loot type cards in your hand. If you purge at least 1, summon a mercenary")
