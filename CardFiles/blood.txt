title("All out Attack")
$Blood=5
$Damage=7
$Range=3
cost(2)
types(blood)
trigger(onPlay,do(unheal($Blood)))
trigger(onPlay,do(damage, ($Damage,(bite),(all),$Range)));
trigger(onPlay,do(damage, ((*,2,$Damage),(bite),(all),(+,-1,$Range))));
text("Lose $Blood health. Deal $Damage damage to all units range $Range, then (getDamage((*,2,$Damage))) damage to all units range (+,-1,$Range)")
removetrigger(endofturn, true,0)
rarity(8)

title("Berserk")

text("When you lose health, gain $Status rage")
