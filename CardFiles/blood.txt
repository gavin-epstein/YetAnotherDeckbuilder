$Damage=8
$Range=3

trigger(onPlay,do(damage, ($Damage,(bite),(all),$Range)));

text(Lose $Blood health. Deal $Damage damage range $Range, then (*,2, $Damage) damage range (+,-1,$Range))
removetrigger(endofturn, true,0)
