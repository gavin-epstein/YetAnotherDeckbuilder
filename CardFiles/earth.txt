title(Barrage)
cost(1)
types(earth, attack)
$Damage=4
$Range=3
text(purge all unplayable cards in hand, deal $Damage damage range $Range for each card purged)
removetrigger(endofturn, true,2)
trigger(onPlay,do(setVar(self,cardnumber,(countModifier(Hand,unplayable)))))
trigger(onPlay,foreach(Loopcard,(select(Hand, (hasModifier(self, unplayable)),"-",all)),(do (purge($Loopcard )))))
trigger(onPlay, if( ($cardnumber>0),(do(damage($Damage,(earth),(any),$Range)))))
trigger(onPlay, if( ($cardnumber>1),(repeat( (do(damage($Damage,(earth),(lastTargets),$Range ))),(+,-1,$cardnumber)))))
removetrigger(endofturn,true,1);
rarity(7);

title(Pyroclastic Flow)

rarity(7);
