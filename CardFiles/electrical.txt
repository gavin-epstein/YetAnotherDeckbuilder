$Charge=2
$Block=12
cost(1)
trigger(onPlay, do(setVar(self, dothething, 0)))
trigger(onPlay, if( (do(deplete($Deplete))), (do(setVar(self, dothething,1)))))
trigger(onPlay, if( ($dothething>0),(do(block($Block)))))
trigger(onPlay, if( ($dothething<1),(do(charge($Charge)))))
text("If you have less than $Deplete battery, charge $Charge, otherwise deplete $Deplete and block $Block")

removetrigger(endofturn, true, 1)
types(electric)

cost(1);

rarity(5);
removetrigger(endofturn, true, 1);
text(Deplete all, deal $Damage damage at any range for each battery depleted);
